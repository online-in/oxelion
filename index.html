class Portfolio:
    def __init__(self, name):
        self.name = name
        self.skills = ['Python', 'JS', 'React']

    def showcase(self):
        # Displaying the developer's skills
        print(f"{self.name}'s Skills:")
        for skill in self.skills:
            print(f"- {skill}")
